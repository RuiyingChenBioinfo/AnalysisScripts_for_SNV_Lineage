### Upset for dataset 1
colnames = ["adata_sc1", "adata_cmb1", "adata_ct1", "adata_pb1"]
adatas = [adata_sc1, adata_cmb1, adata_ct1, adata_pb1]

lists = [pd.Index(a.var_names).astype(str).tolist() for a in adatas]
maxlen = max(map(len, lists))
padded = [lst + [""] * (maxlen - len(lst)) for lst in lists]
varnames_4cols = pd.DataFrame(dict(zip(colnames, padded)))
sets = list(map(set, lists))
all_genes = sorted(set().union(*sets))
memberships = [[name for name, st in zip(colnames, sets) if g in st] for g in all_genes]
s = from_memberships(memberships).reorder_levels(["adata_pb1", "adata_ct1", "adata_cmb1", "adata_sc1"])
plt.figure()
up = UpSet(s, subset_size="count", show_counts=True, sort_by="cardinality", sort_categories_by=None)
up.plot()
#plt.savefig("Backup/SNV_list_under_different_resolution_10x_upset_plot_new.pdf", dpi=300, bbox_inches="tight")
plt.show()
